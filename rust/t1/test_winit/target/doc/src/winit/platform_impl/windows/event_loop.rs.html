97/// some backends like macos uses an uninhabited `Never` type,
98/// on windows, `UserEvent`s are also dispatched through the
99/// WNDPROC callback, and due to the re-entrant nature of the
100/// callback, recursively delivered events must be queued in a
101/// buffer, the current implementation put this queue in
102/// `EventLoopRunner`, which is shared between the event pumping
103/// loop and the callback. because it's hard to decide from the
104/// outside whether a event needs to be buffered, I decided not
105/// use `Event<Never>` for the shared runner state, but use unit
106/// as a placeholder so user events can be buffered as usual,
107/// the real `UserEvent` is pulled from the mpsc channel directly
108/// when the placeholder event is delivered to the event handler

236    pub fn run_on_demand<F>(&mut self, mut event_handler: F) -> Result<(), EventLoopError>
237    where
238        F: FnMut(Event<T>, &RootAEL),
239    {
240        {
241            let runner = &self.window_target.p.runner_shared;
242
243            let event_loop_windows_ref = &self.window_target;
244            let user_event_receiver = &self.user_event_receiver;
245            // # Safety
246            // We make sure to call runner.clear_event_handler() before
247            // returning
248            unsafe {
249                runner.set_event_handler(move |event| {
250                    // the shared `EventLoopRunner` is not parameterized
251                    // `EventLoopProxy::send_event()` calls `PostMessage`
252                    // to wakeup and dispatch a placeholder `UserEvent`,
253                    // when we received the placeholder event here, the
254                    // real UserEvent(T) should already be put in the
255                    // mpsc channel and ready to be pulled
256                    let event = match event.map_nonuser_event() {
257                        Ok(non_user_event) => non_user_event,
258                        Err(_user_event_placeholder) => Event::UserEvent(
259                            user_event_receiver
260                                .try_recv()
261                                .expect("user event signaled but not received"),
262                        ),
263                    };
264                    event_handler(event, event_loop_windows_ref)
265                });
266            }
267        }
268
269        let exit_code = loop {
270            self.wait_for_messages(None);
271            // wait_for_messages calls user application before and after waiting
272            // so it may have decided to exit.
273            if let Some(code) = self.exit_code() {
274                break code;
275            }
276
277            self.dispatch_peeked_messages();
278
279            if let Some(code) = self.exit_code() {
280                break code;
281            }
282        };
283
284        let runner = &self.window_target.p.runner_shared;
285        runner.loop_destroyed();
286
287        // # Safety
288        // We assume that this will effectively call `runner.clear_event_handler()`
289        // to meet the safety requirements for calling `runner.set_event_handler()` above.
290        runner.reset_runner();
291
292        if exit_code == 0 {
293            Ok(())
294        } else {
295            Err(EventLoopError::ExitFailure(exit_code))
296        }
297    }

299    pub fn pump_events<F>(&mut self, timeout: Option<Duration>, mut event_handler: F) -> PumpStatus
300    where
301        F: FnMut(Event<T>, &RootAEL),
302    {
303        {
304            let runner = &self.window_target.p.runner_shared;
305            let event_loop_windows_ref = &self.window_target;
306            let user_event_receiver = &self.user_event_receiver;
307
308            // # Safety
309            // We make sure to call runner.clear_event_handler() before
310            // returning
311            //
312            // Note: we're currently assuming nothing can panic and unwind
313            // to leave the runner in an unsound state with an associated
314            // event handler.
315            unsafe {
316                runner.set_event_handler(move |event| {
317                    let event = match event.map_nonuser_event() {
318                        Ok(non_user_event) => non_user_event,
319                        Err(_user_event_placeholder) => Event::UserEvent(
320                            user_event_receiver
321                                .recv()
322                                .expect("user event signaled but not received"),
323                        ),
324                    };
325                    event_handler(event, event_loop_windows_ref)
326                });
327                runner.wakeup();
328            }
329        }
330
331        if self.exit_code().is_none() {
332            self.wait_for_messages(timeout);
333        }
334        // wait_for_messages calls user application before and after waiting
335        // so it may have decided to exit.
336        if self.exit_code().is_none() {
337            self.dispatch_peeked_messages();
338        }
339
340        let runner = &self.window_target.p.runner_shared;
341
342        let status = if let Some(code) = runner.exit_code() {
343            runner.loop_destroyed();
344
345            // Immediately reset the internal state for the loop to allow
346            // the loop to be run more than once.
347            runner.reset_runner();
348            PumpStatus::Exit(code)
349        } else {
350            runner.prepare_wait();
351            PumpStatus::Continue
352        };
353
354        // We wait until we've checked for an exit status before clearing the
355        // application callback, in case we need to dispatch a LoopExiting event
356        //
357        // # Safety
358        // This pairs up with our call to `runner.set_event_handler` and ensures
359        // the application's callback can't be held beyond its lifetime.
360        runner.clear_event_handler();
361
362        status
363    }

365    /// Waits until new event messages arrive to be peeked.
366    /// Doesn't peek messages itself.
367    ///
368    /// Parameter timeout is optional. This method would wait for the smaller timeout
369    /// between the argument and a timeout from control flow.
370    fn wait_for_messages(&mut self, timeout: Option<Duration>) {
371        let runner = &self.window_target.p.runner_shared;
372
373        // We aim to be consistent with the MacOS backend which has a RunLoop
374        // observer that will dispatch AboutToWait when about to wait for
375        // events, and NewEvents after the RunLoop wakes up.
376        //
377        // We emulate similar behaviour by treating `MsgWaitForMultipleObjectsEx` as our wait
378        // point and wake up point (when it returns) and we drain all other
379        // pending messages via `PeekMessage` until we come back to "wait" via
380        // `MsgWaitForMultipleObjectsEx`.
381        //
382        runner.prepare_wait();
383        wait_for_messages_impl(&mut self.high_resolution_timer, runner.control_flow(), timeout);
384        // Before we potentially exit, make sure to consistently emit an event for the wake up
385        runner.wakeup();
386    }

388    /// Dispatch all queued messages via `PeekMessageW`
389    fn dispatch_peeked_messages(&mut self) {
390        let runner = &self.window_target.p.runner_shared;
391
392        // We generally want to continue dispatching all pending messages
393        // but we also allow dispatching to be interrupted as a means to
394        // ensure the `pump_events` won't indefinitely block an external
395        // event loop if there are too many pending events. This interrupt
396        // flag will be set after dispatching `RedrawRequested` events.
397        runner.interrupt_msg_dispatch.set(false);
398
399        // # Safety
400        // The Windows API has no documented requirement for bitwise
401        // initializing a `MSG` struct (it can be uninitialized memory for the C
402        // API) and there's no API to construct or initialize a `MSG`. This
403        // is the simplest way avoid uninitialized memory in Rust
404        let mut msg: MSG = unsafe { mem::zeroed() };
405
406        loop {
407            unsafe {
408                if PeekMessageW(&mut msg, 0, 0, 0, PM_REMOVE) == false.into() {
409                    break;
410                }
411
412                let handled = if let Some(callback) = self.msg_hook.as_deref_mut() {
413                    callback(&mut msg as *mut _ as *mut _)
414                } else {
415                    false
416                };
417                if !handled {
418                    TranslateMessage(&msg);
419                    DispatchMessageW(&msg);
420                }
421            }
422
423            if let Err(payload) = runner.take_panic_error() {
424                runner.reset_runner();
425                panic::resume_unwind(payload);
426            }
427
428            if let Some(_code) = runner.exit_code() {
429                break;
430            }
431
432            if runner.interrupt_msg_dispatch.get() {
433                break;
434            }
435        }
436    }

546/// Returns the id of the main thread.
547///
548/// Windows has no real API to check if the current executing thread is the "main thread", unlike
549/// macOS.
550///
551/// Windows will let us look up the current thread's id, but there's no API that lets us check what
552/// the id of the main thread is. We would somehow need to get the main thread's id before a
553/// developer could spin off any other threads inside of the main entrypoint in order to emulate the
554/// capabilities of other platforms.
555///
556/// We can get the id of the main thread by using CRT initialization. CRT initialization can be used
557/// to setup global state within a program. The OS will call a list of function pointers which
558/// assign values to a static variable. To have get a hold of the main thread id, we need to place
559/// our function pointer inside of the `.CRT$XCU` section so it is called before the main
560/// entrypoint.
561///
562/// Full details of CRT initialization can be found here:
563/// <https://docs.microsoft.com/en-us/cpp/c-runtime-library/crt-initialization?view=msvc-160>
564fn main_thread_id() -> u32 {
565    static mut MAIN_THREAD_ID: u32 = 0;
566
567    // Function pointer used in CRT initialization section to set the above static field's value.
568
569    // Mark as used so this is not removable.
570    #[used]
571    #[allow(non_upper_case_globals)]
572    // Place the function pointer inside of CRT initialization section so it is loaded before
573    // main entrypoint.
574    //
575    // See: https://doc.rust-lang.org/stable/reference/abi.html#the-link_section-attribute
576    #[link_section = ".CRT$XCU"]
577    static INIT_MAIN_THREAD_ID: unsafe fn() = {
578        unsafe fn initer() {
579            unsafe { MAIN_THREAD_ID = GetCurrentThreadId() };
580        }
581        initer
582    };
583
584    unsafe { MAIN_THREAD_ID }
585}

595fn dur2timeout(dur: Duration) -> u32 {
596    // Note that a duration is a (u64, u32) (seconds, nanoseconds) pair, and the
597    // timeouts in windows APIs are typically u32 milliseconds. To translate, we
598    // have two pieces to take care of:
599    //
600    // * Nanosecond precision is rounded up
601    // * Greater than u32::MAX milliseconds (50 days) is rounded up to INFINITE (never time out).
602    dur.as_secs()
603        .checked_mul(1000)
604        .and_then(|ms| ms.checked_add((dur.subsec_nanos() as u64) / 1_000_000))
605        .and_then(
606            |ms| {
607                if dur.subsec_nanos() % 1_000_000 > 0 {
608                    ms.checked_add(1)
609                } else {
610                    Some(ms)
611                }
612            },
613        )
614        .map(|ms| if ms > u32::MAX as u64 { INFINITE } else { ms as u32 })
615        .unwrap_or(INFINITE)
616}

626/// Set upper limit for waiting time to avoid overflows.
627/// I chose 50 days as a limit because it is used in dur2timeout.
628const FIFTY_DAYS: Duration = Duration::from_secs(50_u64 * 24 * 60 * 60);
629/// Waitable timers use 100 ns intervals to indicate due time.
630/// <https://learn.microsoft.com/en-us/windows/win32/api/synchapi/nf-synchapi-setwaitabletimer#parameters>
631/// And there is no point waiting using other ways for such small timings
632/// because they are even less precise (can overshoot by few ms).
633const MIN_WAIT: Duration = Duration::from_nanos(100);
634
635fn create_high_resolution_timer() -> Option<OwnedHandle> {
636    unsafe {
637        let handle: HANDLE = CreateWaitableTimerExW(
638            ptr::null(),
639            ptr::null(),
640            CREATE_WAITABLE_TIMER_HIGH_RESOLUTION,
641            TIMER_ALL_ACCESS,
642        );
643        // CREATE_WAITABLE_TIMER_HIGH_RESOLUTION is supported only after
644        // Win10 1803 but it is already default option for rustc
645        // (std uses it to implement `std::thread::sleep`).
646        if handle == 0 {
647            None
648        } else {
649            Some(OwnedHandle::from_raw_handle(handle as *mut c_void))
650        }
651    }
652}

654/// This function should not return error if parameters are valid
655/// but there is no guarantee about that at MSDN docs
656/// so we return result of GetLastError if fail.
657///
658/// ## Safety
659///
660/// timer must be a valid timer handle created by [create_high_resolution_timer].
661/// timeout divided by 100 nanoseconds must be more than 0 and less than i64::MAX.
662unsafe fn set_high_resolution_timer(timer: RawHandle, timeout: Duration) -> Result<(), u32> {
663    const INTERVAL_NS: u32 = MIN_WAIT.subsec_nanos();
664    const INTERVALS_IN_SEC: u64 = (Duration::from_secs(1).as_nanos() / INTERVAL_NS as u128) as u64;
665    let intervals_to_wait: u64 =
666        timeout.as_secs() * INTERVALS_IN_SEC + u64::from(timeout.subsec_nanos() / INTERVAL_NS);
667    debug_assert!(intervals_to_wait < i64::MAX as u64, "Must be called with smaller duration",);
668    // Use negative time to indicate relative time.
669    let due_time: i64 = -(intervals_to_wait as i64);
670    unsafe {
671        let set_result = SetWaitableTimer(timer as HANDLE, &due_time, 0, None, ptr::null(), FALSE);
672        if set_result != FALSE {
673            Ok(())
674        } else {
675            Err(GetLastError())
676        }
677    }
678}

680/// Implementation detail of [EventLoop::wait_for_messages].
681///
682/// Does actual system-level waiting and doesn't process any messages itself,
683/// including winits internal notifications about waiting and new messages arrival.
684fn wait_for_messages_impl(
685    high_resolution_timer: &mut Option<OwnedHandle>,
686    control_flow: ControlFlow,
687    timeout: Option<Duration>,
688) {
689    let timeout = {
690        let control_flow_timeout = match control_flow {
691            ControlFlow::Wait => None,
692            ControlFlow::Poll => Some(Duration::ZERO),
693            ControlFlow::WaitUntil(wait_deadline) => {
694                let start = Instant::now();
695                Some(wait_deadline.saturating_duration_since(start))
696            },
697        };
698        let timeout = min_timeout(timeout, control_flow_timeout);
699        if timeout == Some(Duration::ZERO) {
700            // Do not wait if we don't have time.
701            return;
702        }
703        // Now we decided to wait so need to do some clamping
704        // to avoid problems with overflow and calling WinAPI with invalid parameters.
705        timeout
706            .map(|t| t.min(FIFTY_DAYS))
707            // If timeout is less than minimally supported by Windows,
708            // increase it to that minimum. Who want less than microsecond delays anyway?
709            .map(|t| t.max(MIN_WAIT))
710    };
711
712    if timeout.is_some() && high_resolution_timer.is_none() {
713        *high_resolution_timer = create_high_resolution_timer();
714    }
715
716    let high_resolution_timer: Option<RawHandle> =
717        high_resolution_timer.as_ref().map(OwnedHandle::as_raw_handle);
718
719    let use_timer: bool;
720    if let (Some(handle), Some(timeout)) = (high_resolution_timer, timeout) {
721        let res = unsafe {
722            // Safety: handle can be Some only if we succeeded in creating high resolution
723            // timer. We properly clamped timeout so it can be used as argument
724            // to timer.
725            set_high_resolution_timer(handle, timeout)
726        };
727        if let Err(error_code) = res {
728            // We successfully got timer but failed to set it?
729            // Should be some bug in our code.
730            tracing::trace!("Failed to set high resolution timer: last error {}", error_code);
731            use_timer = false;
732        } else {
733            use_timer = true;
734        }
735    } else {
736        use_timer = false;
737    }
738
739    unsafe {
740        // Either:
741        //  1. User wants to wait indefinely if timeout is not set.
742        //  2. We failed to get and set high resolution timer and we need something instead of it.
743        let wait_duration_ms = timeout.map(dur2timeout).unwrap_or(INFINITE);
744
745        let (num_handles, raw_handles) =
746            if use_timer { (1, [high_resolution_timer.unwrap()]) } else { (0, [ptr::null_mut()]) };
747
748        // We must use `QS_ALLINPUT` to wake on accessibility messages.
749        let result = MsgWaitForMultipleObjectsEx(
750            num_handles,
751            raw_handles.as_ptr() as *const _,
752            wait_duration_ms,
753            QS_ALLINPUT,
754            MWMO_INPUTAVAILABLE,
755        );
756        if result == WAIT_FAILED {
757            // Well, nothing smart to do in such case.
758            // Treat it as spurious wake up.
759            tracing::warn!("Failed to MsgWaitForMultipleObjectsEx: error code {}", GetLastError(),);
760        }
761    }
762}

764pub(crate) struct EventLoopThreadExecutor {
765    thread_id: u32,
766    target_window: HWND,
767}

769unsafe impl Send for EventLoopThreadExecutor {}
770unsafe impl Sync for EventLoopThreadExecutor {}

772impl EventLoopThreadExecutor {
773    /// Check to see if we're in the parent event loop's thread.
774    pub(super) fn in_event_loop_thread(&self) -> bool {
775        let cur_thread_id = unsafe { GetCurrentThreadId() };
776        self.thread_id == cur_thread_id
777    }
778
779    /// Executes a function in the event loop thread. If we're already in the event loop thread,
780    /// we just call the function directly.
781    ///
782    /// The `Inserted` can be used to inject a `WindowState` for the callback to use. The state is
783    /// removed automatically if the callback receives a `WM_CLOSE` message for the window.
784    ///
785    /// Note that if you are using this to change some property of a window and updating
786    /// `WindowState` then you should call this within the lock of `WindowState`. Otherwise the
787    /// events may be sent to the other thread in different order to the one in which you set
788    /// `WindowState`, leaving them out of sync.
789    ///
790    /// Note that we use a FnMut instead of a FnOnce because we're too lazy to create an equivalent
791    /// to the unstable FnBox.
792    pub(super) fn execute_in_thread<F>(&self, mut function: F)
793    where
794        F: FnMut() + Send + 'static,
795    {
796        unsafe {
797            if self.in_event_loop_thread() {
798                function();
799            } else {
800                // We double-box because the first box is a fat pointer.
801                let boxed2: ThreadExecFn = Box::new(Box::new(function));
802
803                let raw = Box::into_raw(boxed2);
804
805                let res = PostMessageW(self.target_window, EXEC_MSG_ID.get(), raw as usize, 0);
806                assert!(res != false.into(), "PostMessage failed; is the messages queue full?");
807            }
808        }
809    }
810}

838/// A lazily-initialized window message ID.
839pub struct LazyMessageId {
840    /// The ID.
841    id: AtomicU32,
842
843    /// The name of the message.
844    name: &'static str,
845}
846
847/// An invalid custom window ID.
848const INVALID_ID: u32 = 0x0;
849
850impl LazyMessageId {
851    /// Create a new `LazyId`.
852    const fn new(name: &'static str) -> Self {
853        Self { id: AtomicU32::new(INVALID_ID), name }
854    }
855
856    /// Get the message ID.
857    pub fn get(&self) -> u32 {
858        // Load the ID.
859        let id = self.id.load(Ordering::Relaxed);
860
861        if id != INVALID_ID {
862            return id;
863        }
864
865        // Register the message.
866        // SAFETY: We are sure that the pointer is a valid C string ending with '\0'.
867        assert!(self.name.ends_with('\0'));
868        let new_id = unsafe { RegisterWindowMessageA(self.name.as_ptr()) };
869
870        assert_ne!(
871            new_id,
872            0,
873            "RegisterWindowMessageA returned zero for '{}': {}",
874            self.name,
875            std::io::Error::last_os_error()
876        );
877
878        // Store the new ID. Since `RegisterWindowMessageA` returns the same value for any given
879        // string, the target value will always either be a). `INVALID_ID` or b). the
880        // correct ID. Therefore a compare-and-swap operation here (or really any
881        // consideration) is never necessary.
882        self.id.store(new_id, Ordering::Relaxed);
883
884        new_id
885    }
886}

1049/// Any window whose callback is configured to this function will have its events propagated
1050/// through the events loop of the thread the window was created in.
1051// This is the callback that is called by `DispatchMessage` in the events loop.
1052//
1053// Returning 0 tells the Win32 API that the message has been processed.
1054// FIXME: detect WM_DWMCOMPOSITIONCHANGED and call DwmEnableBlurBehindWindow if necessary
1055pub(super) unsafe extern "system" fn public_window_callback(
1056    window: HWND,
1057    msg: u32,
1058    wparam: WPARAM,
1059    lparam: LPARAM,
1060) -> LRESULT {
1061    let userdata = unsafe { super::get_window_long(window, GWL_USERDATA) };
1062
1063    let userdata_ptr = match (userdata, msg) {
1064        (0, WM_NCCREATE) => {
1065            let createstruct = unsafe { &mut *(lparam as *mut CREATESTRUCTW) };
1066            let initdata = unsafe { &mut *(createstruct.lpCreateParams as *mut InitData<'_>) };
1067
1068            let result = match unsafe { initdata.on_nccreate(window) } {
1069                Some(userdata) => unsafe {
1070                    super::set_window_long(window, GWL_USERDATA, userdata as _);
1071                    DefWindowProcW(window, msg, wparam, lparam)
1072                },
1073                None => -1, // failed to create the window
1074            };
1075
1076            return result;
1077        },
1078        // Getting here should quite frankly be impossible,
1079        // but we'll make window creation fail here just in case.
1080        (0, WM_CREATE) => return -1,
1081        (_, WM_CREATE) => unsafe {
1082            let createstruct = &mut *(lparam as *mut CREATESTRUCTW);
1083            let initdata = createstruct.lpCreateParams;
1084            let initdata = &mut *(initdata as *mut InitData<'_>);
1085
1086            initdata.on_create();
1087            return DefWindowProcW(window, msg, wparam, lparam);
1088        },
1089        (0, _) => return unsafe { DefWindowProcW(window, msg, wparam, lparam) },
1090        _ => userdata as *mut WindowData,
1091    };
1092
1093    let (result, userdata_removed, recurse_depth) = {
1094        let userdata = unsafe { &*(userdata_ptr) };
1095
1096        userdata.recurse_depth.set(userdata.recurse_depth.get() + 1);
1097
1098        let result = unsafe { public_window_callback_inner(window, msg, wparam, lparam, userdata) };
1099
1100        let userdata_removed = userdata.userdata_removed.get();
1101        let recurse_depth = userdata.recurse_depth.get() - 1;
1102        userdata.recurse_depth.set(recurse_depth);
1103
1104        (result, userdata_removed, recurse_depth)
1105    };
1106
1107    if userdata_removed && recurse_depth == 0 {
1108        drop(unsafe { Box::from_raw(userdata_ptr) });
1109    }
1110
1111    result
1112}

1114unsafe fn public_window_callback_inner(
1115    window: HWND,
1116    msg: u32,
1117    wparam: WPARAM,
1118    lparam: LPARAM,
1119    userdata: &WindowData,
1120) -> LRESULT {
1121    let mut result = ProcResult::DefWindowProc(wparam);
1122
1123    // Send new modifiers before sending key events.
1124    let mods_changed_callback = || match msg {
1125        WM_KEYDOWN | WM_SYSKEYDOWN | WM_KEYUP | WM_SYSKEYUP => {
1126            update_modifiers(window, userdata);
1127            result = ProcResult::Value(0);
1128        },
1129        _ => (),
1130    };
1131    userdata
1132        .event_loop_runner
1133        .catch_unwind(mods_changed_callback)
1134        .unwrap_or_else(|| result = ProcResult::Value(-1));
1135
1136    let keyboard_callback = || {
1137        use crate::event::WindowEvent::KeyboardInput;
1138        let events =
1139            userdata.key_event_builder.process_message(window, msg, wparam, lparam, &mut result);
1140        for event in events {
1141            userdata.send_event(Event::WindowEvent {
1142                window_id: RootWindowId(WindowId(window)),
1143                event: KeyboardInput {
1144                    device_id: DEVICE_ID,
1145                    event: event.event,
1146                    is_synthetic: event.is_synthetic,
1147                },
1148            });
1149        }
1150    };
1151    userdata
1152        .event_loop_runner
1153        .catch_unwind(keyboard_callback)
1154        .unwrap_or_else(|| result = ProcResult::Value(-1));
1155
1156    // I decided to bind the closure to `callback` and pass it to catch_unwind rather than passing
1157    // the closure to catch_unwind directly so that the match body indentation wouldn't change and
1158    // the git blame and history would be preserved.
1159    let callback = || match msg {
1160        WM_NCCALCSIZE => {
1161            let window_flags = userdata.window_state_lock().window_flags;
1162            if wparam == 0 || window_flags.contains(WindowFlags::MARKER_DECORATIONS) {
1163                result = ProcResult::DefWindowProc(wparam);
1164                return;
1165            }
1166
1167            let params = unsafe { &mut *(lparam as *mut NCCALCSIZE_PARAMS) };
1168
1169            if util::is_maximized(window) {
1170                // Limit the window size when maximized to the current monitor.
1171                // Otherwise it would include the non-existent decorations.
1172                //
1173                // Use `MonitorFromRect` instead of `MonitorFromWindow` to select
1174                // the correct monitor here.
1175                // See https://github.com/MicrosoftEdge/WebView2Feedback/issues/2549
1176                let monitor = unsafe { MonitorFromRect(&params.rgrc[0], MONITOR_DEFAULTTONULL) };
1177                if let Ok(monitor_info) = monitor::get_monitor_info(monitor) {
1178                    params.rgrc[0] = monitor_info.monitorInfo.rcWork;
1179                }
1180            } else if window_flags.contains(WindowFlags::MARKER_UNDECORATED_SHADOW) {
1181                // Extend the client area to cover the whole non-client area.
1182                // https://docs.microsoft.com/en-us/windows/win32/winmsg/wm-nccalcsize#remarks
1183                //
1184                // HACK(msiglreith): To add the drop shadow we slightly tweak the non-client area.
1185                // This leads to a small black 1px border on the top. Adding a margin manually
1186                // on all 4 borders would result in the caption getting drawn by the DWM.
1187                //

2367                    // shared and nudging away from the wrong monitor based on those.
2368                    #[allow(clippy::bool_to_int_with_if)]
2369                    let delta_nudge_to_dpi_monitor = (
2370                        if wrong_monitor_rect.left == new_monitor_rect.right {
2371                            -1
2372                        } else if wrong_monitor_rect.right == new_monitor_rect.left {
2373                            1
2374                        } else {
2375                            0
2376                        },
2377                        if wrong_monitor_rect.bottom == new_monitor_rect.top {
2378                            1
2379                        } else if wrong_monitor_rect.top == new_monitor_rect.bottom {
2380                            -1
2381                        } else {
2382                            0
2383                        },
2384                    );
2385
2386                    let abort_after_iterations = new_monitor_rect.right - new_monitor_rect.left
2387                        + new_monitor_rect.bottom
2388                        - new_monitor_rect.top;
2389                    for _ in 0..abort_after_iterations {
2390                        conservative_rect.left += delta_nudge_to_dpi_monitor.0;
2391                        conservative_rect.right += delta_nudge_to_dpi_monitor.0;
2392                        conservative_rect.top += delta_nudge_to_dpi_monitor.1;
2393                        conservative_rect.bottom += delta_nudge_to_dpi_monitor.1;
2394
2395                        if unsafe { MonitorFromRect(&conservative_rect, MONITOR_DEFAULTTONULL) }
2396                            == new_dpi_monitor
2397                        {
2398                            break;
2399                        }
2400                    }
2401
2402                    conservative_rect
2403                };
2404            }
2405
2406            unsafe {
2407                SetWindowPos(
2408                    window,
2409                    0,
2410                    new_outer_rect.left,
2411                    new_outer_rect.top,
2412                    new_outer_rect.right - new_outer_rect.left,
2413                    new_outer_rect.bottom - new_outer_rect.top,
2414                    SWP_NOZORDER | SWP_NOACTIVATE,
2415                )
2416            };
2417
2418            result = ProcResult::Value(0);
2419        },
2420
2421        WM_SETTINGCHANGE => {
2422            use crate::event::WindowEvent::ThemeChanged;
2423
2424            let preferred_theme = userdata.window_state_lock().preferred_theme;
2425
2426            if preferred_theme.is_none() {
2427                let new_theme = try_theme(window, preferred_theme);
2428                let mut window_state = userdata.window_state_lock();
2429
2430                if window_state.current_theme != new_theme {
2431                    window_state.current_theme = new_theme;
2432                    drop(window_state);
2433                    userdata.send_event(Event::WindowEvent {
2434                        window_id: RootWindowId(WindowId(window)),
2435                        event: ThemeChanged(new_theme),
2436                    });
2437                }
2438            }
2439            result = ProcResult::DefWindowProc(wparam);
2440        },
2441
2442        _ => {
2443            if msg == DESTROY_MSG_ID.get() {
2444                unsafe { DestroyWindow(window) };
2445                result = ProcResult::Value(0);
2446            } else if msg == SET_RETAIN_STATE_ON_SIZE_MSG_ID.get() {
2447                let mut window_state = userdata.window_state_lock();
2448                window_state.set_window_flags_in_place(|f| {
2449                    f.set(WindowFlags::MARKER_RETAIN_STATE_ON_SIZE, wparam != 0)
2450                });
2451                result = ProcResult::Value(0);
2452            } else if msg == TASKBAR_CREATED.get() {
2453                let window_state = userdata.window_state_lock();
2454                unsafe { set_skip_taskbar(window, window_state.skip_taskbar) };
2455                result = ProcResult::DefWindowProc(wparam);
2456            } else {
2457                result = ProcResult::DefWindowProc(wparam);
2458            }
2459        },
2460    };
2461
2462    userdata
2463        .event_loop_runner
2464        .catch_unwind(callback)
2465        .unwrap_or_else(|| result = ProcResult::Value(-1));
2466
2467    match result {
2468        ProcResult::DefWindowProc(wparam) => unsafe { DefWindowProcW(window, msg, wparam, lparam) },
2469        ProcResult::Value(val) => val,
2470    }
2471}

2473unsafe extern "system" fn thread_event_target_callback(
2474    window: HWND,
2475    msg: u32,
2476    wparam: WPARAM,
2477    lparam: LPARAM,
2478) -> LRESULT {
2479    let userdata_ptr =
2480        unsafe { super::get_window_long(window, GWL_USERDATA) } as *mut ThreadMsgTargetData;
2481    if userdata_ptr.is_null() {
2482        // `userdata_ptr` will always be null for the first `WM_GETMINMAXINFO`, as well as
2483        // `WM_NCCREATE` and `WM_CREATE`.
2484        return unsafe { DefWindowProcW(window, msg, wparam, lparam) };
2485    }
2486    let userdata = unsafe { Box::from_raw(userdata_ptr) };
2487
2488    if msg != WM_PAINT {
2489        unsafe { RedrawWindow(window, ptr::null(), 0, RDW_INTERNALPAINT) };
2490    }
2491
2492    let mut userdata_removed = false;
2493
2494    // I decided to bind the closure to `callback` and pass it to catch_unwind rather than passing
2495    // the closure to catch_unwind directly so that the match body indentation wouldn't change and
2496    // the git blame and history would be preserved.
2497    let callback = || match msg {
2498        WM_NCDESTROY => {
2499            unsafe { super::set_window_long(window, GWL_USERDATA, 0) };
2500            userdata_removed = true;
2501            0
2502        },
2503        WM_PAINT => unsafe {
2504            ValidateRect(window, ptr::null());
2505            // Default WM_PAINT behaviour. This makes sure modals and popups are shown immediately
2506            // when opening them.
2507            DefWindowProcW(window, msg, wparam, lparam)
2508        },
2509
2510        WM_INPUT_DEVICE_CHANGE => {
2511            let event = match wparam as u32 {
2512                GIDC_ARRIVAL => DeviceEvent::Added,
2513                GIDC_REMOVAL => DeviceEvent::Removed,
2514                _ => unreachable!(),
2515            };
2516
2517            userdata
2518                .send_event(Event::DeviceEvent { device_id: wrap_device_id(lparam as u32), event });
2519
2520            0
2521        },
2522
2523        WM_INPUT => {
2524            if let Some(data) = raw_input::get_raw_input_data(lparam as _) {
2525                unsafe { handle_raw_input(&userdata, data) };
2526            }
2527
2528            unsafe { DefWindowProcW(window, msg, wparam, lparam) }
2529        },
2530
2531        _ if msg == USER_EVENT_MSG_ID.get() => {
2532            // synthesis a placeholder UserEvent, so that if the callback is
2533            // re-entered it can be buffered for later delivery. the real
2534            // user event is still in the mpsc channel and will be pulled
2535            // once the placeholder event is delivered to the wrapper
2536            // `event_handler`
2537            userdata.send_event(Event::UserEvent(UserEventPlaceholder));
2538            0
2539        },
2540        _ if msg == EXEC_MSG_ID.get() => {
2541            let mut function: ThreadExecFn = unsafe { Box::from_raw(wparam as *mut _) };
2542            function();
2543            0
2544        },
2545        _ => unsafe { DefWindowProcW(window, msg, wparam, lparam) },
2546    };
2547
2548    let result = userdata.event_loop_runner.catch_unwind(callback).unwrap_or(-1);
2549    if userdata_removed {
2550        drop(userdata);
2551    } else {
2552        Box::leak(userdata);
2553    }
2554    result
2555}

2557unsafe fn handle_raw_input(userdata: &ThreadMsgTargetData, data: RAWINPUT) {
2558    use crate::event::DeviceEvent::{Button, Key, Motion, MouseMotion, MouseWheel};
2559    use crate::event::ElementState::{Pressed, Released};
2560    use crate::event::MouseScrollDelta::LineDelta;
2561
2562    let device_id = wrap_device_id(data.header.hDevice as _);
2563
2564    if data.header.dwType == RIM_TYPEMOUSE {
2565        let mouse = unsafe { data.data.mouse };
2566
2567        if util::has_flag(mouse.usFlags as u32, MOUSE_MOVE_RELATIVE) {
2568            let x = mouse.lLastX as f64;
2569            let y = mouse.lLastY as f64;
2570
2571            if x != 0.0 {
2572                userdata.send_event(Event::DeviceEvent {
2573                    device_id,
2574                    event: Motion { axis: 0, value: x },
2575                });
2576            }
2577
2578            if y != 0.0 {
2579                userdata.send_event(Event::DeviceEvent {
2580                    device_id,
2581                    event: Motion { axis: 1, value: y },
2582                });
2583            }
2584
2585            if x != 0.0 || y != 0.0 {
2586                userdata.send_event(Event::DeviceEvent {
2587                    device_id,
2588                    event: MouseMotion { delta: (x, y) },
2589                });
2590            }
2591        }
2592
2593        let button_flags = unsafe { mouse.Anonymous.Anonymous.usButtonFlags };
2594        if util::has_flag(button_flags as u32, RI_MOUSE_WHEEL) {
2595            let button_data = unsafe { mouse.Anonymous.Anonymous.usButtonData } as i16;
2596            let delta = button_data as f32 / WHEEL_DELTA as f32;
2597            userdata.send_event(Event::DeviceEvent {
2598                device_id,
2599                event: MouseWheel { delta: LineDelta(0.0, delta) },
2600            });
2601        }
2602        if util::has_flag(button_flags as u32, RI_MOUSE_HWHEEL) {
2603            let button_data = unsafe { mouse.Anonymous.Anonymous.usButtonData } as i16;
2604            let delta = -button_data as f32 / WHEEL_DELTA as f32;
2605            userdata.send_event(Event::DeviceEvent {
2606                device_id,
2607                event: MouseWheel { delta: LineDelta(delta, 0.0) },
2608            });
2609        }
2610
2611        let button_state = raw_input::get_raw_mouse_button_state(button_flags as u32);
2612        for (button, state) in button_state.iter().enumerate() {
2613            if let Some(state) = *state {
2614                userdata.send_event(Event::DeviceEvent {
2615                    device_id,
2616                    event: Button { button: button as _, state },
2617                });
2618            }
2619        }
2620    } else if data.header.dwType == RIM_TYPEKEYBOARD {
2621        let keyboard = unsafe { data.data.keyboard };
2622
2623        let pressed = keyboard.Message == WM_KEYDOWN || keyboard.Message == WM_SYSKEYDOWN;
2624        let released = keyboard.Message == WM_KEYUP || keyboard.Message == WM_SYSKEYUP;
2625
2626        if !pressed && !released {
2627            return;
2628        }
2629
2630        if let Some(physical_key) = raw_input::get_keyboard_physical_key(keyboard) {
2631            let state = if pressed { Pressed } else { Released };
2632
2633            userdata.send_event(Event::DeviceEvent {
2634                device_id,
2635                event: Key(RawKeyEvent { physical_key, state }),
2636            });
2637        }
2638    }
2639}

2641enum PointerMoveKind {
2642    /// Pointer enterd to the window.
2643    Enter,
2644    /// Pointer leaved the window client area.
2645    Leave,
2646    /// Pointer is inside the window or `GetClientRect` failed.

2650fn get_pointer_move_kind(
2651    window: HWND,
2652    mouse_was_inside_window: bool,
2653    x: i32,
2654    y: i32,
2655) -> PointerMoveKind {
2656    let rect: RECT = unsafe {
2657        let mut rect: RECT = mem::zeroed();
2658        if GetClientRect(window, &mut rect) == false.into() {
2659            return PointerMoveKind::None; // exit early if GetClientRect failed
2660        }
2661        rect
2662    };
2663
2664    let x = (rect.left..rect.right).contains(&x);
2665    let y = (rect.top..rect.bottom).contains(&y);
2666
2667    if !mouse_was_inside_window && x && y {
2668        PointerMoveKind::Enter
2669    } else if mouse_was_inside_window && !(x && y) {
2670        PointerMoveKind::Leave
2671    } else {
2672        PointerMoveKind::None
2673    }
