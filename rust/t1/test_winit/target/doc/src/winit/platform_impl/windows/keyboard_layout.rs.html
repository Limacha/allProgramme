97    pub struct WindowsModifiers : u8 {
98        const SHIFT = 1 << 0;
99        const CONTROL = 1 << 1;
100        const ALT = 1 << 2;
101        const CAPS_LOCK = 1 << 3;
102        const FLAGS_END = 1 << 4;
103    }
104}

106impl WindowsModifiers {
107    pub fn active_modifiers(key_state: &[u8; 256]) -> WindowsModifiers {
108        let shift = key_state[VK_SHIFT as usize] & 0x80 != 0;
109        let lshift = key_state[VK_LSHIFT as usize] & 0x80 != 0;
110        let rshift = key_state[VK_RSHIFT as usize] & 0x80 != 0;
111
112        let control = key_state[VK_CONTROL as usize] & 0x80 != 0;
113        let lcontrol = key_state[VK_LCONTROL as usize] & 0x80 != 0;
114        let rcontrol = key_state[VK_RCONTROL as usize] & 0x80 != 0;
115
116        let alt = key_state[VK_MENU as usize] & 0x80 != 0;
117        let lalt = key_state[VK_LMENU as usize] & 0x80 != 0;
118        let ralt = key_state[VK_RMENU as usize] & 0x80 != 0;
119
120        let caps = key_state[VK_CAPITAL as usize] & 0x01 != 0;
121
122        let mut result = WindowsModifiers::empty();
123        if shift || lshift || rshift {
124            result.insert(WindowsModifiers::SHIFT);
125        }
126        if control || lcontrol || rcontrol {
127            result.insert(WindowsModifiers::CONTROL);
128        }
129        if alt || lalt || ralt {
130            result.insert(WindowsModifiers::ALT);
131        }
132        if caps {
133            result.insert(WindowsModifiers::CAPS_LOCK);
134        }
135
136        result
137    }
138
139    pub fn apply_to_kbd_state(self, key_state: &mut [u8; 256]) {
140        if self.intersects(Self::SHIFT) {
141            key_state[VK_SHIFT as usize] |= 0x80;
142        } else {
143            key_state[VK_SHIFT as usize] &= !0x80;
144            key_state[VK_LSHIFT as usize] &= !0x80;
145            key_state[VK_RSHIFT as usize] &= !0x80;
146        }
147        if self.intersects(Self::CONTROL) {
148            key_state[VK_CONTROL as usize] |= 0x80;
149        } else {
150            key_state[VK_CONTROL as usize] &= !0x80;
151            key_state[VK_LCONTROL as usize] &= !0x80;
152            key_state[VK_RCONTROL as usize] &= !0x80;
153        }
154        if self.intersects(Self::ALT) {
155            key_state[VK_MENU as usize] |= 0x80;
156        } else {
157            key_state[VK_MENU as usize] &= !0x80;
158            key_state[VK_LMENU as usize] &= !0x80;
159            key_state[VK_RMENU as usize] &= !0x80;
160        }
161        if self.intersects(Self::CAPS_LOCK) {
162            key_state[VK_CAPITAL as usize] |= 0x01;
163        } else {
164            key_state[VK_CAPITAL as usize] &= !0x01;
165        }
166    }
167
168    /// Removes the control modifier if the alt modifier is not present.
169    /// This is useful because on Windows: (Control + Alt) == AltGr
170    /// but we don't want to interfere with the AltGr state.
171    pub fn remove_only_ctrl(mut self) -> WindowsModifiers {
172        if !self.contains(WindowsModifiers::ALT) {
173            self.remove(WindowsModifiers::CONTROL);
174        }
175        self
176    }

179pub(crate) struct Layout {
180    pub hkl: u64,
181
182    /// Maps numpad keys from Windows virtual key to a `Key`.
183    ///
184    /// This is useful because some numpad keys generate different characters based on the locale.
185    /// For example `VK_DECIMAL` is sometimes "." and sometimes ",". Note: numpad-specific virtual
186    /// keys are only produced by Windows when the NumLock is active.
187    ///
188    /// Making this field separate from the `keys` field saves having to add NumLock as a modifier
189    /// to `WindowsModifiers`, which would double the number of items in keys.
190    pub numlock_on_keys: HashMap<VIRTUAL_KEY, Key>,
191    /// Like `numlock_on_keys` but this will map to the key that would be produced if numlock was
192    /// off. The keys of this map are identical to the keys of `numlock_on_keys`.
193    pub numlock_off_keys: HashMap<VIRTUAL_KEY, Key>,
194
195    /// Maps a modifier state to group of key strings
196    /// We're not using `ModifiersState` here because that object cannot express caps lock,
197    /// but we need to handle caps lock too.
198    ///
199    /// This map shouldn't need to exist.
200    /// However currently this seems to be the only good way
201    /// of getting the label for the pressed key. Note that calling `ToUnicode`
202    /// just when the key is pressed/released would be enough if `ToUnicode` wouldn't
203    /// change the keyboard state (it clears the dead key). There is a flag to prevent
204    /// changing the state, but that flag requires Windows 10, version 1607 or newer)
205    pub keys: HashMap<WindowsModifiers, HashMap<KeyCode, Key>>,
206    pub has_alt_graph: bool,
207}
208
209impl Layout {
210    pub fn get_key(
211        &self,
212        mods: WindowsModifiers,
213        num_lock_on: bool,
214        vkey: VIRTUAL_KEY,
215        physical_key: &PhysicalKey,
216    ) -> Key {
217        let native_code = NativeKey::Windows(vkey);
218
219        let unknown_alt = vkey == VK_MENU;
220        if !unknown_alt {
221            // Here we try using the virtual key directly but if the virtual key doesn't distinguish
222            // between left and right alt, we can't report AltGr. Therefore, we only do this if the
223            // key is not the "unknown alt" key.
224            //
225            // The reason for using the virtual key directly is that `MapVirtualKeyExW` (used when
226            // building the keys map) sometimes maps virtual keys to odd scancodes that don't match
227            // the scancode coming from the KEYDOWN message for the same key. For example: `VK_LEFT`
228            // is mapped to `0x004B`, but the scancode for the left arrow is `0xE04B`.
229            let key_from_vkey =
230                vkey_to_non_char_key(vkey, native_code.clone(), self.hkl, self.has_alt_graph);
231
232            if !matches!(key_from_vkey, Key::Unidentified(_)) {
233                return key_from_vkey;
234            }
235        }
236        if num_lock_on {
237            if let Some(key) = self.numlock_on_keys.get(&vkey) {
238                return key.clone();
239            }
240        } else if let Some(key) = self.numlock_off_keys.get(&vkey) {
241            return key.clone();
242        }
243        if let PhysicalKey::Code(code) = physical_key {
244            if let Some(keys) = self.keys.get(&mods) {
245                if let Some(key) = keys.get(code) {
246                    return key.clone();
247                }
248            }
249        }
250        Key::Unidentified(native_code)
251    }
252}

260impl LayoutCache {
261    /// Checks whether the current layout is already known and
262    /// prepares the layout if it isn't known.
263    /// The current layout is then returned.
264    pub fn get_current_layout(&mut self) -> (u64, &Layout) {
265        let locale_id = unsafe { GetKeyboardLayout(0) } as u64;
266        match self.layouts.entry(locale_id) {
267            Entry::Occupied(entry) => (locale_id, entry.into_mut()),
268            Entry::Vacant(entry) => {
269                let layout = Self::prepare_layout(locale_id);
270                (locale_id, entry.insert(layout))
271            },
272        }
273    }
274
275    pub fn get_agnostic_mods(&mut self) -> ModifiersState {
276        let (_, layout) = self.get_current_layout();
277        let filter_out_altgr = layout.has_alt_graph && key_pressed(VK_RMENU);
278        let mut mods = ModifiersState::empty();
279        mods.set(ModifiersState::SHIFT, key_pressed(VK_SHIFT));
280        mods.set(ModifiersState::CONTROL, key_pressed(VK_CONTROL) && !filter_out_altgr);
281        mods.set(ModifiersState::ALT, key_pressed(VK_MENU) && !filter_out_altgr);
282        mods.set(ModifiersState::SUPER, key_pressed(VK_LWIN) || key_pressed(VK_RWIN));
283        mods
284    }

286    fn prepare_layout(locale_id: u64) -> Layout {
287        let mut layout = Layout {
288            hkl: locale_id,
289            numlock_on_keys: Default::default(),
290            numlock_off_keys: Default::default(),
291            keys: Default::default(),
292            has_alt_graph: false,
293        };
294
295        // We initialize the keyboard state with all zeros to
296        // simulate a scenario when no modifier is active.
297        let mut key_state = [0u8; 256];
298
299        // `MapVirtualKeyExW` maps (non-numpad-specific) virtual keys to scancodes as if numlock
300        // was off. We rely on this behavior to find all virtual keys which are not numpad-specific
301        // but map to the numpad.
302        //
303        // src_vkey: VK  ==>  scancode: u16 (on the numpad)
304        //
305        // Then we convert the source virtual key into a `Key` and the scancode into a virtual key
306        // to get the reverse mapping.
307        //
308        // src_vkey: VK  ==>  scancode: u16 (on the numpad)
309        //    ||                    ||
310        //    \/                    \/
311        // map_value: Key  <-  map_vkey: VK
312        layout.numlock_off_keys.reserve(NUMPAD_KEYCODES.len());
313        for vk in 0..256 {
314            let scancode = unsafe { MapVirtualKeyExW(vk, MAPVK_VK_TO_VSC_EX, locale_id as HKL) };
315            if scancode == 0 {
316                continue;
317            }
318            let keycode = match scancode_to_physicalkey(scancode) {
319                PhysicalKey::Code(code) => code,
320                // TODO: validate that we can skip on unidentified keys (probably never occurs?)
321                _ => continue,
322            };
323            if !is_numpad_specific(vk as VIRTUAL_KEY) && NUMPAD_KEYCODES.contains(&keycode) {
324                let native_code = NativeKey::Windows(vk as VIRTUAL_KEY);
325                let map_vkey = keycode_to_vkey(keycode, locale_id);
326                if map_vkey == 0 {
327                    continue;
328                }
329                let map_value =
330                    vkey_to_non_char_key(vk as VIRTUAL_KEY, native_code, locale_id, false);
331                if matches!(map_value, Key::Unidentified(_)) {
332                    continue;
333                }
334                layout.numlock_off_keys.insert(map_vkey, map_value);
335            }
336        }
337
338        layout.numlock_on_keys.reserve(NUMPAD_VKEYS.len());
339        for vk in NUMPAD_VKEYS.iter() {
340            let vk = (*vk) as u32;
341            let scancode = unsafe { MapVirtualKeyExW(vk, MAPVK_VK_TO_VSC_EX, locale_id as HKL) };
342            let unicode = Self::to_unicode_string(&key_state, vk, scancode, locale_id);
343            if let ToUnicodeResult::Str(s) = unicode {
344                layout.numlock_on_keys.insert(vk as VIRTUAL_KEY, Key::Character(SmolStr::new(s)));
345            }
346        }
347
348        // Iterate through every combination of modifiers
349        let mods_end = WindowsModifiers::FLAGS_END.bits();
350        for mod_state in 0..mods_end {
351            let mut keys_for_this_mod = HashMap::with_capacity(256);
352
353            let mod_state = WindowsModifiers::from_bits_retain(mod_state);
354            mod_state.apply_to_kbd_state(&mut key_state);
355
356            // Virtual key values are in the domain [0, 255].
357            // This is reinforced by the fact that the keyboard state array has 256
358            // elements. This array is allowed to be indexed by virtual key values
359            // giving the key state for the virtual key used for indexing.
360            for vk in 0..256 {
361                let scancode =
362                    unsafe { MapVirtualKeyExW(vk, MAPVK_VK_TO_VSC_EX, locale_id as HKL) };
363                if scancode == 0 {
364                    continue;
365                }
366
367                let native_code = NativeKey::Windows(vk as VIRTUAL_KEY);
368                let key_code = match scancode_to_physicalkey(scancode) {
369                    PhysicalKey::Code(code) => code,
370                    // TODO: validate that we can skip on unidentified keys (probably never occurs?)
371                    _ => continue,
372                };
373                // Let's try to get the key from just the scancode and vk
374                // We don't necessarily know yet if AltGraph is present on this layout so we'll
375                // assume it isn't. Then we'll do a second pass where we set the "AltRight" keys to
376                // "AltGr" in case we find out that there's an AltGraph.
377                let preliminary_key =
378                    vkey_to_non_char_key(vk as VIRTUAL_KEY, native_code, locale_id, false);
379                match preliminary_key {
380                    Key::Unidentified(_) => (),
381                    _ => {
382                        keys_for_this_mod.insert(key_code, preliminary_key);
383                        continue;
384                    },
385                }
386
387                let unicode = Self::to_unicode_string(&key_state, vk, scancode, locale_id);
388                let key = match unicode {
389                    ToUnicodeResult::Str(str) => Key::Character(SmolStr::new(str)),
390                    ToUnicodeResult::Dead(dead_char) => {
391                        // println!("{:?} - {:?} produced dead {:?}", key_code, mod_state,
392                        // dead_char);
393                        Key::Dead(dead_char)
394                    },
395                    ToUnicodeResult::None => {
396                        let has_alt = mod_state.contains(WindowsModifiers::ALT);
397                        let has_ctrl = mod_state.contains(WindowsModifiers::CONTROL);
398                        // HACK: `ToUnicodeEx` seems to fail getting the string for the numpad
399                        // divide key, so we handle that explicitly here
400                        if !has_alt && !has_ctrl && key_code == KeyCode::NumpadDivide {
401                            Key::Character(SmolStr::new("/"))
402                        } else {
403                            // Just use the unidentified key, we got earlier
404                            preliminary_key
405                        }
406                    },
407                };
408
409                // Check for alt graph.
410                // The logic is that if a key pressed with no modifier produces
411                // a different `Character` from when it's pressed with CTRL+ALT then the layout
412                // has AltGr.
413                let ctrl_alt: WindowsModifiers = WindowsModifiers::CONTROL | WindowsModifiers::ALT;
414                let is_in_ctrl_alt = mod_state == ctrl_alt;
415                if !layout.has_alt_graph && is_in_ctrl_alt {
416                    // Unwrapping here because if we are in the ctrl+alt modifier state
417                    // then the alt modifier state must have come before.
418                    let simple_keys = layout.keys.get(&WindowsModifiers::empty()).unwrap();
419                    if let Some(Key::Character(key_no_altgr)) = simple_keys.get(&key_code) {
420                        if let Key::Character(key) = &key {
421                            layout.has_alt_graph = key != key_no_altgr;
422                        }
423                    }
424                }
425
426                keys_for_this_mod.insert(key_code, key);
427            }
428            layout.keys.insert(mod_state, keys_for_this_mod);
429        }
430
431        // Second pass: replace right alt keys with AltGr if the layout has alt graph
432        if layout.has_alt_graph {
433            for mod_state in 0..mods_end {
434                let mod_state = WindowsModifiers::from_bits_retain(mod_state);
435                if let Some(keys) = layout.keys.get_mut(&mod_state) {
436                    if let Some(key) = keys.get_mut(&KeyCode::AltRight) {
437                        *key = Key::Named(NamedKey::AltGraph);
438                    }
439                }
440            }
441        }
442
443        layout
444    }

446    fn to_unicode_string(
447        key_state: &[u8; 256],
448        vkey: u32,
449        scancode: u32,
450        locale_id: u64,
451    ) -> ToUnicodeResult {
452        unsafe {
453            let mut label_wide = [0u16; 8];
454            let mut wide_len = ToUnicodeEx(
455                vkey,
456                scancode,
457                (&key_state[0]) as *const _,
458                (&mut label_wide[0]) as *mut _,
459                label_wide.len() as i32,
460                0,
461                locale_id as HKL,
462            );
463            if wide_len < 0 {
464                // If it's dead, we run `ToUnicode` again to consume the dead-key
465                wide_len = ToUnicodeEx(
466                    vkey,
467                    scancode,
468                    (&key_state[0]) as *const _,
469                    (&mut label_wide[0]) as *mut _,
470                    label_wide.len() as i32,
471                    0,
472                    locale_id as HKL,
473                );
474                if wide_len > 0 {
475                    let os_string = OsString::from_wide(&label_wide[0..wide_len as usize]);
476                    if let Ok(label_str) = os_string.into_string() {
477                        if let Some(ch) = label_str.chars().next() {
478                            return ToUnicodeResult::Dead(Some(ch));
479                        }
480                    }
481                }
482                return ToUnicodeResult::Dead(None);
483            }
484            if wide_len > 0 {
485                let os_string = OsString::from_wide(&label_wide[0..wide_len as usize]);
486                if let Ok(label_str) = os_string.into_string() {
487                    return ToUnicodeResult::Str(label_str);
488                }
489            }
490        }
491        ToUnicodeResult::None
492    }
493}
